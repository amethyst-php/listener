Result

Once you've got the result you should always check if an error has occurred, if not, retrieve the resource.

use Railken\Amethyst\Managers\TemplateManager;

$manager = new TemplateManager();

$result = $manager->create([
    "name" => "a common name0.58049600 1562549199",
    "filename" => "test.pdf",
    "filetype" => "application/pdf",
    "description" => "A description",
    "content" => "The cake is a {{ message }}",
    "data_builder" => [
        "name" => "Florida Wisozk",
        "description" => "Aliquam delectus eligendi vel aut expedita perspiciatis quia est. Ut fuga omnis cum eius recusandae fuga aut. Inventore ducimus necessitatibus voluptas consequatur voluptatem et.",
        "class_name" => "Railken\\Amethyst\\DataBuilders\\CommonDataBuilder",
        "class_arguments" => "- Railken\\Amethyst\\Managers\\DataBuilderManager\n",
        "filter" => "id eq 1",
        "input" => "date:\n    type: date\n    validation: 'date_format:Y-m-d'\n",
        "mock_data" => "date: '2018-01-01'\n",
        "include" => "bar"
    ]
]);

if ($result->ok()) {

    $resource = $result->getResource();

} else {

    // Loop through all errors
    $result->getErrors()->map(function($error) {
        return $error->toArray();
    }))

    // Retrieve an array of all errors
    $result->getSimpleErrors();

    /* The result is something like this:

        [0] => Array
            (
                [code] => FIELD_NOT_DEFINED
                [label] => field
                [message] => The field is required
                [value] =>
            )
    */

}
