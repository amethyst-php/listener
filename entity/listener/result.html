Result

Once you've got the result you should always check if an error has occurred, if not, retrieve the resource.

use Railken\Amethyst\Managers\ListenerManager;

$manager = new ListenerManager();

$result = $manager->create([
    "name" => "El. psy. congroo. 1562549199.5955",
    "condition" => "{{ message is not empty ? 1 : 0 }}",
    "work" => [
        "name" => "Dr. Kaylie Homenick",
        "payload" => "class: Railken\\Amethyst\\Workers\\EmailWorker\ndata:\n    id: 1\n",
        "description" => "Eaque et facilis nihil sit quisquam. At ad fugiat et eligendi ad. Illum qui voluptatibus omnis sed odit quo id.",
        "data_builder" => [
            "name" => "Aniya Jerde",
            "description" => "Distinctio voluptatum quisquam odio tenetur qui fugiat voluptas. Quam nam rerum doloremque nulla. Sapiente dolorum voluptates enim rerum aliquam repellendus.",
            "class_name" => "Railken\\Amethyst\\DataBuilders\\CommonDataBuilder",
            "class_arguments" => "- Railken\\Amethyst\\Managers\\DataBuilderManager\n",
            "filter" => "id eq 1",
            "input" => "date:\n    type: date\n    validation: 'date_format:Y-m-d'\n",
            "mock_data" => "date: '2018-01-01'\n",
            "include" => "bar",
            "data_builder" => [
                "class_name" => "Railken\\Amethyst\\DataBuilders\\DummyDataBuilder"
            ]
        ],
        "data" => "id: '{{ record.id }}'\n"
    ],
    "data" => "dummy1: dummy2\n",
    "event_class" => "Dummy"
]);

if ($result->ok()) {

    $resource = $result->getResource();

} else {

    // Loop through all errors
    $result->getErrors()->map(function($error) {
        return $error->toArray();
    }))

    // Retrieve an array of all errors
    $result->getSimpleErrors();

    /* The result is something like this:

        [0] => Array
            (
                [code] => FIELD_NOT_DEFINED
                [label] => field
                [message] => The field is required
                [value] =>
            )
    */

}
